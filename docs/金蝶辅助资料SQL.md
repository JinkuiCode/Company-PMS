select 
A.FNUMBER TypeCode
,B.FNAME  TypeName
,C.FNUMBER ProjectCode
,D.FDATAVALUE ProjectName
,C.FENTRYID

from T_BAS_ASSISTANTDATA A
left join T_BAS_ASSISTANTDATA_L B on A.FID=B.FID and B.FLOCALEID='2052'
left join T_BAS_ASSISTANTDATAENTRY C on A.FID=C.FID
left join T_BAS_ASSISTANTDATAENTRY_L D on C.FENTRYID=D.FENTRYID and D.FLOCALEID='2052'

where B.FNAME ='销售项目'
